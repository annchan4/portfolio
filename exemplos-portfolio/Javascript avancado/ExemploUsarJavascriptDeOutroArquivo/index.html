Fazer arquivo js usar coisas de outro arquivo

Basicamente, tenho um arquivo scripts.js que vai importar algumas coisas que um arquivo calc.js vai exportar

Inspecione o console para ver o resultado

calc.js:

        
export var myVar = 5;

export function myFunc()
{
    console.log('Imprimi algo no console');
}

export class Calc
{
    static sum(a,b)
    {
        return a + b;
    }
}

scripts.js:

        
import {Calc, myVar, myFunc} from './calc.js';

myFunc();
console.log(Calc.sum(5,3),myVar);
