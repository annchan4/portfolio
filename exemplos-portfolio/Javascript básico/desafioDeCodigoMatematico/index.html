function calculo(a,b,n)
{
    var resposta = "";
    for (var i = 0; i < n; i++)
    {
        var somaDosDoisElevadoAAlgo = 0;
        for (var j = 0; j <= i; j++)
        {
            somaDosDoisElevadoAAlgo = somaDosDoisElevadoAAlgo + (2**j) * b;
        }

    
        var maisUm = a + somaDosDoisElevadoAAlgo;
        resposta = resposta + maisUm + " ";
    }

    return resposta;
}
