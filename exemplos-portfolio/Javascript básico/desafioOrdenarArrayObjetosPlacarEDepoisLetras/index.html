function auxOrdPlacarENomes(a,b)
{
    //dado um array, ordena inversamente por placar, mas caso tenha 2 itens
    //de mesmo placar, ordena alfabeticamente
    if(a.placar < b.placar)
    {
        return 1;
    }
    else if(a.placar == b.placar && a.nome > b.nome)
    {
        return 1;
    }
    else
    {
        return -1;
    }
}

function imprimirArrayNomesEPlacar(array1)
{
    for(var i = 0; i < array1.length; i++)
    {
        console.log(array1[i].nome + " " + array1[i].placar);
    }
}
